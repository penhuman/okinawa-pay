        function formatMoney(amount) {

            return new Intl.NumberFormat('zh-TW').format(amount);

        }


        // 日期格式化

        function formatDate(timestamp) {

            if (!timestamp) return '剛剛';

            const date = timestamp.toDate ? timestamp.toDate() : new Date(timestamp);

            const now = new Date();

            const isToday = date.getDate() === now.getDate() && date.getMonth() === now.getMonth() && date.getFullYear() === now.getFullYear();

            

            if (isToday) {

                return `今天 ${date.getHours().toString().padStart(2, '0')}:${date.getMinutes().toString().padStart(2, '0')}`;

            }

            

            return `${date.getMonth() + 1}/${date.getDate()} ${date.getHours().toString().padStart(2, '0')}:${date.getMinutes().toString().padStart(2, '0')}`;

        }

        function setupRealtimeListener() {

            if (!currentUser) return;


            // 注意：為了共享帳本，使用 public data collection

            const expensesRef = collection(db, 'artifacts', appId, 'public', 'data', 'expenses');

            

            // 這裡不使用 orderBy 或 where，因為需要建立索引，我們改在客戶端排序

            const q = query(expensesRef);


            unsubscribeSnapshot = onSnapshot(q, (snapshot) => {

                allExpenses = [];

                snapshot.forEach((doc) => {

                    allExpenses.push({ id: doc.id, ...doc.data() });

                });


                // 在客戶端進行排序 (依時間遞減)

                allExpenses.sort((a, b) => {

                    const timeA = a.createdAt ? (a.createdAt.toMillis ? a.createdAt.toMillis() : a.createdAt) : 0;

                    const timeB = b.createdAt ? (b.createdAt.toMillis ? b.createdAt.toMillis() : b.createdAt) : 0;

                    return timeB - timeA; // 新的在上面

                });


                els.loadingSpinner.style.display = 'none';

                renderDashboard();

                renderExpenseList();

            }, (error) => {

                console.error("讀取資料失敗:", error);

                showToast("讀取資料失敗", "error");

            });

        }


        // 計算並渲染統計數據

        function renderDashboard() {

            let total = 0;

            let fridaTotal = 0;

            let tedTotal = 0;

            let xingxuanTotal = 0;

            const rate = Number(els.exchangeRate.value) || 0.21;


            allExpenses.forEach(expense => {

                const amt = Number(expense.amount) || 0;

                const cur = expense.currency || 'TWD';

                const amtInTWD = cur === 'JPY' ? amt * rate : amt;


                total += amtInTWD;

                if (expense.payer === 'Frida') {

                    fridaTotal += amtInTWD;

                } else if (expense.payer === 'Husband' || expense.payer === 'Ted') {

                    tedTotal += amtInTWD;

                } else if (expense.payer === 'Xingxuan') {

                    xingxuanTotal += amtInTWD;

                }

            });


            els.totalAmount.textContent = formatMoney(Math.round(total));

            els.fridaPaid.textContent = `$${formatMoney(Math.round(fridaTotal))}`;

            els.tedPaid.textContent = `$${formatMoney(Math.round(tedTotal))}`;

            els.xingxuanPaid.textContent = `$${formatMoney(Math.round(xingxuanTotal))}`;


            // 計算結算金額 (假設花費由 3 人平分)

            const perPerson = total / 3;

            els.settlementHint.classList.remove('hidden');

            

            if (total === 0) {

                 els.settlementHint.classList.add('hidden');

            } else {

                const balances = [

                    { name: 'Frida', balance: fridaTotal - perPerson },

                    { name: 'Ted', balance: tedTotal - perPerson },

                    { name: '幸萱', balance: xingxuanTotal - perPerson }

                ];

                

                let html = `<div class="text-gray-600 mb-2 font-bold border-b border-gray-200 pb-1">平均每人應付: $${formatMoney(Math.round(perPerson))}</div>`;

                

                balances.forEach(b => {

                    const diff = Math.round(b.balance);

                    if (diff > 0) {

                        html += `<div class="mt-1">${b.name} <span class="text-green-600 font-bold">應收回 $${formatMoney(diff)}</span></div>`;

                    } else if (diff < 0) {

                        html += `<div class="mt-1">${b.name} <span class="text-red-500 font-bold">需再拿 $${formatMoney(Math.abs(diff))}</span></div>`;

                    } else {

                        html += `<div class="mt-1">${b.name} <span class="text-gray-500">已結清</span></div>`;

                    }

                });

                

                els.settlementHint.innerHTML = html;

                els.settlementHint.className = "mt-4 text-sm font-medium bg-white p-4 rounded-xl text-left border border-gray-200 shadow-sm";

            }

        }

        async function saveExpense() {

            if (!currentUser) return;

            

            const amount = Number(els.amountInput.value);

            const title = els.titleInput.value.trim();

            const note = els.noteInput.value.trim();

            const payer = document.querySelector('input[name="payer"]:checked').value;

            const currency = document.querySelector('input[name="currency"]:checked').value;

                els.amountInput.focus();

                return;

            }

                showToast("請輸入名稱", "error");

                els.titleInput.focus();

                return;

                currency,

                    await addDoc(collectionRef, expenseData);

            }

        }
